3类循环
比如打印1-10的数字
While循环
declare
  pnum number:=1;
begin
  while punm<10 loop
     dbms_output.put_line(pnum);
     pnum:=pnum+1;
  end loop;
end;

Loop循环
declare
  pnum number:=1;
begin
  loop
   exit when punm>10;
dbms_output.put_line(pnum);
     pnum:=pnum+1;
  end loop
end

For循环
declare
  pnum number:=1;
begin
  for punm in 1..10 loop
  dbms_output.put_line(pnum);
  pnum:=pnum+1;
end
